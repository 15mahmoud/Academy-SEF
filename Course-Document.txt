  - Course Name

  - Course Instructor
      [string - it cannot be null]

  - Course Level
      [number - it cannot be null]

  - Course Language
      [string - it cannot be null]

  - Start Date of Course
      [string - it cannot be null]

  - Course Duration
      [string  - it cannot be null]

  - Course Certificate
      [boolean - default is false - it can be null]

  - Course Introduction
      [string  - it cannot be null]

  - Course Assessment
      [string  - it cannot be null]

  - Course Requirements
      [string  - it cannot be null]

  - Course Materials
      [string  - it cannot be null]

  - Course Publishing Date
      [string  - it cannot be null]
